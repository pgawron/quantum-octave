 2  ## @deftypefn {Function file} {} Dec2BinVec(@var{number},@var{length})
 3  ## Function @code{Dec2BinVec} returns binary vector representing
 4  ## @var{LENGTH}-bit representation of number @var{number}. This
 5  ## number must be nonegative ang less then 2^@var{len}.
 6  ## 
 7  ## @example 
 8  ## @group
 9  ## Dev2BinVec(4,3)
10  ##  @result{} 
11  ##  1 0 0
12  ## Dev2BinVec(4,4)
13  ##  @result{} 
14  ##  0 1 0 0
15  ## @end group
16  ## @end example
17  ##
18  ## @end deftypefn
19  ##
20  ## @seealso {dec2bin, BuildBinaryVector}
21  ##
22  ## Author: Piotr Gawron, Jaroslaw Miszczak

24  ## Last modyfication: 15 March 2004
25
26  function ret = Dec2BinVec(number, len)
27  if (nargin != 2)
28      error("Two numbers required!");
29  endif
30  # number must be nonegative
31  if( number > 2^len )
32      error("Number %d is to big to fit length %d!", number, len);
33  endif
34
35  # get number as a string
36  temp = dec2bin([number,2^len-1])(1,:);
37
38  # build binary vector based on number
39  ret = zeros(1,len);
40  # fill it!
41  for i=1:len
42      ret(i) = str2num(temp(i));
43  endfor
44
45  endfunction
