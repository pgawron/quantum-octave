 2  ## @deftypefn {Function file} {} Dec2BinVec(@var{number},@var{length})
 3  ## Function @code{Dec2BinVec} returns binary vector representing
 4  ## @var{LENGTH}-bit representation of number @var{number}. This
 5  ## number must be nonegative ang less then 2^@var{len}.
 6  ## 
 7  ## @example 
 8  ## @group
 9  ## Dev2BinVec(4,3)
10  ##  @result{} 
11  ##  1 0 1
12  ## Dev2BinVec(4,4)
13  ##  @result{} 
14  ##  1 0 1 0
15  ## @end group
16  ## @end example
17  ##
18  ## @end deftypefn
19  ##
20  ## @seealso {dec2bin, BuildBinaryVector}
21  ##

26  function ret = Dec2BinVec(number, len)
27  # number must be nonegative
28  if( number > 2^len )
29      error("Number %d is to big to fit length %d!", number, len);
30  endif
31
32  # get number as a string
33  temp = dec2bin([number,2^len-1])(1,:);
34
35  # build binary vector based on number
36  ret = zeros(1,len);
37  # fill it!
38  for i=1:len
39      ret(i) = str2num(temp(i));
40  endfor
41
42  endfunction
