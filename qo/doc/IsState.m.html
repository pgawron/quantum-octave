1  ## -*- texinfo -*-
 2  ## @deftypefn {Function file} {} IsState(@var{mtx},@var{verb})
 3  ## Function @code{IsPPT} returns @code{1} if matrix @var{mtx} is 
 4  ## vailid density matrix and @code{0} if it is not. If second argument 
 5  ## is @code{verb}
 6  ## 
 7  ## @example 
 8  ## @group
 9  ## IsPPT(State(Ket([0,0,0])))
10  ##  @result{} 
11  ##  1
12  ## @end group
13  ## @end example
14  ##
15  ## @end deftypefn
16  ##
17  ## @seealso {State, MixStates}
18  ##
19  ## Author: Piotr Gawron, Jaroslaw Miszczak
20  ## Created: 22 March 2004
21
22  function ret = IsState(mtx, verb)
23  norm = false;
24  sadj = false;
25  posi = false;
26  why1 = "";
27  why2 = "";
28  why3 = "";
29
30  if (nargin > 2 && nargin < 1)
31      usage ("IsState (state,[verb])");
32  else
33      # test if tr(mtx) = 1
34      if ( trace(mtx) == 1 )
35          norm = true;
36      else
37          why1 = "not normalize";
38      endif
39      # test if mtx = mtx'
40      if ( mtx == mtx')
41          sadj = true;
42      else
43          why2 = "not selfadjoint";
44      endif
45      # test if matrix is ppositive
46      if ( min(eig(mtx)) < 0 )
47          why3 = "not positive definde";
48      else
49          posi = true;
50      endif
51  endif
52
53  if ( norm && sadj &&posi )
54      ret = true;
55  elseif  ( nargin == 2 && verb == 1 )
56      printf("This matrix is not vaild state: is %s %s %s\n",why1,why2,why3);
57      ret =  false;
58  else
59      ret = false;
60  endif
61  endfunction
