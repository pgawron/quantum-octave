1  #Params in:     dotvek - vector of indexes of qubits on which Not operation should be performed
 2  #       regsize - size of register on which Not operation is performed 
 3  #Params out:    matrix of size regsize^2 x regsize^2
 4
 5  function ret = NotBig(dotvek,regsize)
 6  idx = 1;
 7  sort(dotvek);
 8  if (dotvek(1) == 1)
 9      tmp = [0,1;1,0];
10      idx++;
11  else
12      tmp = Id(1);
13  endif
14
15
16  for i = 2:log2(regsize)
17  if((idx <= length(dotvek)) && (dotvek(idx) == i))
18      tmp = kron(tmp,[0,1;1,0]);
19      idx++;
20  else
21      tmp = kron(tmp,Id(1));
22  endif
23
24  endfor
25  ret = tmp;
26  endfunction
