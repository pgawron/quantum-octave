 2  ## @deftypefn {Function file} {} ControledGate(@var{size},@var{gate},@var{contrv},@var{targetv})
 3  ## Function @code{ControledGate} returns controled operation @var{gate} on qubits @var{targetv} 
 4  ## controled by @var{contrv}. Size of the result gate is given by @var{size}.
 5  ## @example 
 6  ## @group
 7  ##  ControledGate (2,Not,[1],[2])
 8  ##  @result{} 
 9  ##  1  0  0  0  
10  ##  0  1  0  0  
11  ##  0  0  0  1
12  ##  0  0  1  0  
13  ## @end group
14  ## @end example
15  ##
16  ## @end deftypefn
17  ##
18  ## @seealso {Id, Not, H, RotX, RotY, RotZ}
19  ##
20  ## Author: Piotr Gawron, Jaroslaw Miszczak
21  ##
22  ## Created: 25 November 2003
23  ##
24  ## Last modification : 10 February 2004
25
26  function ret = ControledGate(size,gate,contrv,targetv)
27
28  if(nargin!=4)
29      usage("ControledGate(size,gate,contrv,targetv)\n");
30  endif
31  if(!is_scalar(size))
32      error("1st parameter should be scalar");
33  endif
34  if(is_square(gate)!=2)
35      error("2nd parameter should be matrix 2x2");
36  endif
37
38  if(size < max(max(contrv),max(targetv)))
39      error("Operator acts on %d qubits, max control index is %d, max target index is %d", size, max(contrv), max(targetv));
40  endif
41
42  proj1 = [0,0;0,1];
43
44  sort(contrv);
45  sort(targetv);
46
47  flags = zeros(1,size);
48
49  for i = 1:length(contrv)
50      ctmp = contrv(i);
51      flags(ctmp) = 1;
52  endfor
53
54  for i = 1:length(targetv)
55      ttmp = targetv(i);
56      if flags(ttmp) != 0
57          error("arguments overlaping");
58          return;
59      else
60      flags(ttmp) = 2;
61      endif
62  endfor
63
64  # initialization
65
66  ret = Id(size);
67
68  if (flags(1) == 1)
69      tmp = proj1;
70  elseif (flags(1) == 2)
71      tmp = Id(1) - gate;
72  elseif (flags(1) == 0)
73      tmp = Id(1);
74  else
75      error("unknown error occured");
76  endif
77
78  # matrix building
79
80  for i = 2:size
81  if (flags(i) == 1)
82      tmp = kron(tmp,proj1);
83  elseif (flags(i) == 2)
84      tmp = kron(tmp,Id(1) - gate);
85  elseif (flags(i) == 0)
86      tmp = kron(tmp,Id(1));
87  else
88      error("unknown error occured");
89  endif
90  endfor
91
92  tmp = Id(size) - tmp;
93
94  ret = tmp;
95
96  endfunction
