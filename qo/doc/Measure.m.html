26  function ret = Measure(state)
27    state_size = size(state)(1);
28  #  qubit_count = log2(state_size)
29
30    ret = zeros(2,state_size);
31    ret(1,:) = 0:state_size-1;
32    temp = zeros(state_size);
33
34    temp(1,1)=1;
35    ret(2,1)=trace(temp*state);
36
37    for i=2:state_size
38    temp(i-1,i-1)=0;
39    temp(i,i)=1;
40
41    ret(2,i)=trace(temp*state);
42    endfor
43   ret = ret';
44  endfunction
