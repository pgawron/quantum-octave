1  ## -*- texinfo -*-
 2  ## @deftypefn {Function File} Normalize (@var{Ket})
 3  ## The @code{Normalize} function generates normalized Ket vector from any
 4  ## number of @var{Ket}s. 
 5  ## @example
 6  ## Normalize (Ket([0,1,1]),2*Ket([1,0,0]))
 7  ## @result{}
 8  ##  0.00000
 9  ##  0.00000
10  ##  0.00000
11  ##  0.44721
12  ##  0.89443
13  ##  0.00000
14  ##  0.00000
15  ##  0.00000
16  ## @end example
17  ## @end deftypefn
18  ## @seealso{State, Evolve}
19  ## Author: Piotr Gawron, Jaroslaw Miszczak
20  ## Created: 25 November 2003
21
22  function ret = Normalize(varargin)
23  ret = [0];
24  vek = va_arg();
25  tempmtr = zeros(length(vek),nargin);
26
27  va_start();
28  ketlngth = length(vek);
29  while (nargin--)
30      vek = va_arg();
31      if ( isvector (vek) && length(vek)==ketlngth)
32          tempmtr(:,nargin+1)=vek;
33      else
34          error("Normalize: vectors are of different sizes");
35          clear ret;
36          return;
37      endif
38  endwhile
39
40  ret = sum(tempmtr,2);
41  norm = sqrt(ret'*ret);
42  ret = ret./norm;
43  endfunction
