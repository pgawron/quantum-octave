 2  ## @deftypefn {Function File} {} MixStates (@var{state}, @dots{})
 3  ## This function allow to represent a mixture of mixed states.
 4  ## The argument is a list of density matrices (for example returned
 5  ## by function State). Every density matrix can be preceded by a
 6  ## (complex) number. If no number is specifed the default
 7  ## value 1 is used. If a numbe is specified then it is
 8  ## normalied so that the result would be vaild density matrix.
 9  ## For example:
10  ## @example
11  ## @group
12  ## MixState(State(Ket([0,1])), 0.5, State(Ket([1,0])))
13  ##  @result{}
14  ## 0.00000  0.00000  0.00000  0.00000
15  ## 0.00000  0.66667  0.00000  0.00000
16  ## 0.00000  0.00000  0.33333  0.00000
17  ## 0.00000  0.00000  0.00000  0.00000
18  ## @end group
19  ## @end example
20  ## @end deftypefn
21
22  ## Author: Piotr Gawron, Jaroslaw Miszczak
23  ## Created: 25 November 2003
24
25  function ret = MixStates(varargin)
26  summtr = 0;
27  coef = 1;
28  sum = 0;
29
30  matrix_size = [0,0];
31
32  while(nargin--)
33  temp = va_arg();
34      if(length(temp)>1)
35          if (coef==1)
36                  sum++;
37          endif
38
39          if(matrix_size==[0,0])
40              matrix_size = size(temp);
41          elseif(matrix_size!=size(temp))
42              error("Matrices have to be the same size");
43          elseif(size(temp)(1)!=size(temp)(2))
44              error("Matrices have to be square");
45          elseif(floor(log2(size(temp)))!=ceil(log2(size(temp))))
46              error("Matrices have to be size 2^n x 2^n");
47          endif
48
49          summtr+=coef.*temp;
50          coef=1;
51      else
52          coef=temp;
53          sum+=coef;
54      endif
55  endwhile
56  sum;
57  ret = summtr./sum;
58
59  endfunction
