 2  ## @deftypefn {Function File} {} Ket (@var{binary_vector}, @dots{})
 3  ## The @code{Ket} function generates vertical complex vector from any
 4  ## number of @var{binary_vector}s.
 5  ##
 6  ## @example
 7  ## @group
 8  ## Ket([1,0,1])
 9  ## @result{}
10  ##  0
11  ##  0
12  ##  0
13  ##  0
14  ##  0
15  ##  1
16  ##  0
17  ##  0
18  ## @end group
19  ## @end example
20  ##
21  ## @end deftypefn

28  function ret = Ket (varargin)
29  a = [1,0]';
30  b = [0,1]';
31  ret = [0];
32  tempvek = 0;
33
34  vek = va_arg();
35  nargin--;
36  while (nargin--)
37      vek = [vek, va_arg()];
38  endwhile
39
40  if ( isvector (vek))
41      if (vek(1) == 0)
42          tempvek = a;
43      elseif (vek(1) == 1)
44          tempvek = b;
45      else
46          error("%d is other than 0 or 1", vek(1));
47      endif
48      for i = 2:length (vek);
49          if (vek(i) == 0)
50              tempvek = kron(tempvek,a);
51          elseif (vek(i) == 1)
52              tempvek = kron(tempvek,b);
53          else
54              error("%d is other than 0 or 1", vek(i));
55          endif
56      endfor
57  endif
58  ret = tempvek;
59  endfunction
