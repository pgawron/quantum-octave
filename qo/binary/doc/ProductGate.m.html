 2  ## @deftypefn {Function file} {} ProductGate(@var{size},@var{gate},@var{targetv})
 3  ## Function @code{ProductGate} generates operator of size equal 2^@var{size}.
 4  ## Parameter @var{gate} is 2x2 basic operator (ex. Not). Paramter
 5  ## @var{targetv} is list of indices of target qubits.
 6  ##
 7  ## @example 
 8  ## @group
 9  ## ProductGate(3,Not,[1,2])
10  ##  @result{}
11  ## 0  0  0  0  0  0  1  0
12  ## 0  0  0  0  0  0  0  1
13  ## 0  0  0  0  1  0  0  0
14  ## 0  0  0  0  0  1  0  0
15  ## 0  0  1  0  0  0  0  0
16  ## 0  0  0  1  0  0  0  0
17  ## 1  0  0  0  0  0  0  0
18  ## 0  1  0  0  0  0  0  0
19  ##
20  ## @end group
21  ## @end example
22  ##
23  ## @end deftypefn
24  ##
25  ## @seealso {ControledGate, Id, Not, H, Pase}
26  ##
27  ## Author: Piotr Gawron, Jaroslaw Miszczak
28  ## Created: 25 November 2003
29
30  function ret = ProductGate(size,gate,targetv)
31  if(nargin!=3)
32      usage("ProductGate(size,gate,targetv)");
33  endif
34  if(!is_scalar(size))
35      error("1st parameter should be scalar");
36  endif
37  if(is_square(gate)!=2)
38      error("2nd parameter should be matrix 2x2");
39  endif
40  if(size < max(targetv))
41      error("Operator acts on %d qubits, max target index is %d", size, max(targetv));
42  endif
43  if (min(targetv)<1)
44      error("Qubit index less than 0");
45  endif
46
47  idx = 1;
48  sort(targetv);
49  if (targetv(1) == 1)
50      tmp = gate;
51      idx++;
52  else
53      tmp = Id(1);
54  endif
55
56  for i = 2:size
57  if((idx <= length(targetv)) && (targetv(idx) == i))
58      tmp = kron(tmp,gate);
59      idx++;
60  else
61      tmp = kron(tmp,Id(1));
62  endif
63
64  endfor
65  ret = tmp;
66  endfunction
