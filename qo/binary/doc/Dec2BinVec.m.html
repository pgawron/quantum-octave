 2  ## @deftypefn {Function file} {} Dec2BinVec(@var{number},@var{length})
 3  ## Function @code{Dec2BinVec} returns binary vector representing
 4  ## @var{length}-bit representation of number @var{number}. This
 5  ## number must be nonegative ang less then 2^@var{len}.
 6  ## 
 7  ## @example 
 8  ## @group
 9  ## Dev2BinVec(4,3)
10  ##  @result{} 
11  ##  1 0 0
12  ## Dev2BinVec(4,4)
13  ##  @result{} 
14  ##  0 1 0 0
15  ## @end group
16  ## @end example
17  ##
18  ## @end deftypefn
19  ##
20  ## @seealso {dec2bin, BinVec2Dec, BuildBinaryVector}
21  ##
22  ## Author: Piotr Gawron, Jaroslaw Miszczak
23  ## Created: 8 March 2004
24  ## Last modyfication: 22 March 2004
25
26  function ret = Dec2BinVec(number, len)
27  if (nargin != 2)
28      usage("Two numbers required!");
29  endif
30
31  # number must be nonegative
32  if( number > 2^len )
33      error("Number %d is to big to fit length %d!", number, len);
34  endif
35
36  # get number as a string
37  temp = dec2bin([number,2^len-1])(1,:);
38
39  # build binary vector based on number
40  ret = zeros(1,len);
41  # fill it!
42  for i=1:len
43      ret(i) = str2num(temp(i));
44  endfor
45  endfunction
