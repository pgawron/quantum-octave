1  ## -*- texinfo -*-
 2  ## @deftypefn {Function File} {} GetPureState(@var{state})
 3  ## The @code{GetPureState} function returns pure state coresponding 
 4  ## to density matrix @var{state}. If @var{state} cannot be decompose
 5  ## in unique way, it returns @code{NaN}.
 6  ##
 7  ## @example
 8  ## @group
 9  ## GetPureState(MaxMix(2))
10  ## @result{}
11  ##  NaN
12  ## @end group
13  ## @end example
14  ## @end deftypefn
15  ## @seealso{Entropy, Id, H, Phase}
16  ##
17  ## Author: Piotr Gawron, Jaroslaw Miszczak
18  ## Created: 26 March 2004
19
20  function ret = GetPureState(state)
21  if ( nargin != 1 )
22      usage("GetPureState (mixed_state)");
23  endif
24      [v,l] = eig(state);
25      sl = size(l)(2);
26      if( l(sl, sl) - 1 < 0.000)
27          ret = v(:,size(v)(2));
28      else
29          ret = NaN;
30      endif
31  endfunction
