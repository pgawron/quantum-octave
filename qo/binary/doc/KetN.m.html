 3  ## The @code{Ket} function generates vertical complex vector from
 4  ## nonnegative number @var{int} in @var{size}-qubit space. If @var{size} is

25  function ret = KetN (number,size)
26  if(nargin==1)
27      lsize = ceil(log2(number+1));
28  elseif(nargin==2)
29      lsize = size;
30  else
31      error("KetN takes no more than 2 parameters");
32  endif
33
34
35  if(number>2^lsize)
36      error("Number too big to fit into vector space");
37  endif
38
39  ret = zeros(2^lsize,1);
40  ret(number+1)=1;
41
42  endfunction
