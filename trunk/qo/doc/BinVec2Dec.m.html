1  ## -*- texinfo -*-
 2  ## @deftypefn {Function file} {} BinVec2Dec(@var{binvec})
 3  ## Function @code{BinVec2Dec} returns number build form binary column 
 4  ## vector @code{binvec = [b1,..,bn]} according to the formula
 5  ##
 6  ##  ret = sum(2.^[0:bs-1].*binvec');
 7  ##
 8  ## wher @code{bs} is dimension of @var{binvec}.
 9  ##
10  ## NOTE: @code{i}-th element of @var{binvec} represents @code{i-1}-th
11  ## bit of number. For example
12  ## 
13  ## @example 
14  ## @group
15  ## BinVec2Dec([1,0,0,0])
16  ##  @result{} 
17  ##  8
18  ## @end group
19  ## @end example
20  ##
21  ## @end deftypefn
22  ##
23  ## @seealso {dec2bin, BinVec2Dec, BuildBinaryVector}
24  ##
25  ## Author: Piotr Gawron, Jaroslaw Miszczak
26  ## Created: 8 March 2004
27  ## Last modyfication: 22 March 2004
28  function ret = BinVec2Dec (binvec)
29  if (nargin != 1)
30      usage ("BinVec2Dec (binary_vector)");
31  endif
32
33  if ( IsBinVec(binvec) != 1 )
34      error ("Binary vector expected!");
35  endif
36
37  bs = size(binvec)(2);
38
39  # reverse binvec
40  for i = 1:bs
41      temp(i) = binvec(bs+1 - i);
42  endfor
43
44  # build number based on binary vector
45  ret = sum(2.^[0:bs-1].*temp');
46
47  endfunction
