 2  ## @deftypefn {Function file} {} CNot(@var{size},@var{contrv},@var{targetv})
 3  ## Function @code{CNot} returns controled Not operation on qubits @var{targetv} 
 4  ## controled by @var{contrv}. Size of the result gate is given by @var{size}.
 5  ## @example 
 6  ## @group
 7  ##  CNot (2,[2],[1])
 8  ##  @result{} 
 9  ##  1  0  0  0  
10  ##  0  0  0  1  
11  ##  0  0  1  0
12  ##  0  1  0  0  
13  ## @end group
14  ## @end example
15  ##
16  ## @end deftypefn
17  ##
18  ## @seealso {Not, ControledGate}

26  function ret = CNot(size,contrv,targetv)
27
28  if(nargin!=3)
29      usage("CNot(size,contrv,targetv)\n");
30  endif
31  if(!is_scalar(size))
32      error("1st parameter should be scalar");
33  endif
34
35  if(size < max(max(contrv),max(targetv)))
36      error("Operator acts on %d qubits, max control index is %d, max target index is %d", size, max(contrv), max(targetv));
37  endif
38
39  proj1 = [0,0;0,1];
40
41  sort(contrv);
42  sort(targetv);
43
44  flags = zeros(1,size);
45
46  for i = 1:length(contrv)
47      ctmp = contrv(i);
48      flags(ctmp) = 1;
49  endfor
50
51  for i = 1:length(targetv)
52      ttmp = targetv(i);
53      if flags(ttmp) != 0
54          error("arguments overlaping");
55          return;
56      else
57      flags(ttmp) = 2;
58      endif
59  endfor
60
61  # initialization
62
63  ret = Id(size);
64
65  if (flags(1) == 1)
66      tmp = proj1;
67  elseif (flags(1) == 2)
68      tmp = Id(1) - Not;
69  elseif (flags(1) == 0)
70      tmp = Id(1);
71  else
72      error("unknown error occured");
73  endif
74
75  # matrix building
76
77  for i = 2:size
78  if (flags(i) == 1)
79      tmp = kron(tmp,proj1);
80  elseif (flags(i) == 2)
81      tmp = kron(tmp,Id(1) - Not);
82  elseif (flags(i) == 0)
83      tmp = kron(tmp,Id(1));
84  else
85      error("unknown error occured");
86  endif
87  endfor
88
89  tmp = Id(size) - tmp;
90
91  ret = tmp;
92
93  endfunction
