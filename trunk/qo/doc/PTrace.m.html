 2  ## @deftypefn {Function file} {} PTrance(state, targv)
 3  ## Function @code{PTrace} returns density matrix obtained
 4  ## from matrix @var{state} by tracing out subspace of qubits
 5  ## listed in @var{targv}.
 6  ##
 7  ## @example 
 8  ## @group
 9  ## PTrace(State(Ket([0,0,0])),[1,2])
10  ##  @result{} 
11  ##  1 0
12  ##  0 0
13  ## @end group
14  ## @end example
15  ##
16  ## @end deftypefn
17  ##
18  ## @seealso {ProductGate, PartialTranspose}
19  ##
20  ## Author: Piotr Gawron, Jaroslaw Miszczak
21  ## Created: 15 March 2004
22
23  function ret = PTrace(state,tqidx)
24  targv = sort(tqidx); # must be sorted!
25
26  ss = size(state)(2); # dimension of matrix state - always power of 2
27  sq = log2(ss); # numer of qubits in matrix state
28
29  tq = size(targv)(2); # number of qubits to trace-out
30  ts = 2^tq;
31
32  rq = sq - tq; # calculate number of qubits in returned density matrix
33  rs = 2^rq; # calculate size of returned density matrix
34
35  ret = zeros (rs); # inicialize returned density matrix
36
37  # loop over every matrix emelent of returned density matrix
38  for i = 1:rs
39      for j = 1:rs
40  #*      printf("(%d,%d) ",i,j);
41  #*      printf("[%d], [%d]\n",Dec2BinVec(i-1,rq),Dec2BinVec(j-1,rq));
42          bi = Dec2BinVec(i-1,rq);
43          bj = Dec2BinVec(j-1,rq);
44
45      # sum loop over elements of targv list
46          for k = 0:ts-1
47              bk = Dec2BinVec(k,tq);
48  #*          printf("k  = %d\n", b);
49              BRA = BuildBinaryVector(targv,bk,bi,sq)';
50              KET = BuildBinaryVector(targv,bk,bj,sq);
51              temp = BRA*(state*KET);
52              ret(i,j) +=  temp;
53          endfor
54      endfor
55  endfor
56  endfunction
