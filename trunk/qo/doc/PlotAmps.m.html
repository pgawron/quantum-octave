1  ## -*- texinfo -*-
 2  ## @deftypefn {Function file} {} PlotAmps(@var{ket},@var{type})
 3  ## Function @code{PlotAmps} plots amplitude of @var{ket} vector in standard base. Second argument describes which part of amplitude is plotted. "Re" - only real part, "Im" - only imaginary part, "RI" - real and imaginary part with diffrent colors. If the second is ommited it is assumed to be "Re".
 4  ## Try this:
 5  ## @example 
 6  ##  @group
 7  ## PlotAmps(Ket([1,1,0]))
 8  ## PlotAmps(Ket([1,0])+Ket([0,1]),"Im")
 9  ## PlotAmps(Ket([1,0])+i*Ket([0,1]),"RI")
10  ## @end group
11  ## @end example
12  ##
13  ## @end deftypefn
14  ##
15  ## @seealso {PlotProbs, Measure}
16  ##
17  ## Author: Piotr Gawron, Jaroslaw Miszczak
18  ## Created: 4 February 2004
19
20  function PlotAmps(varargin)
21
22  # TODO need checking if ket is valid
23  # first argument is allways ket
24  # if we have only one parameter then we draw real part
25
26  if (nargin==1)
27      temp = va_arg();
28      if(!isvector(temp))
29          error("Argument should be ket vector");
30      else
31          plot([0:length(temp)-1],real(temp),"^;Amplitudes;");
32      endif
33  elseif (nargin==2)
34      temp = va_arg();
35      if(!isvector(temp))
36          error("Argument should be ket vector");
37      else
38          ReIm = va_arg();
39          if(ReIm=="Re")
40              plot([0:length(temp)-1],real(temp),"^;Real;");
41          elseif(ReIm=="Im")
42              plot([0:length(temp)-1],imag(temp),"^;Imag;");
43          elseif(ReIm=="RI")
44              plot([0:length(temp)-1],real(temp),"^;Real;", [0:length(temp)-1],imag(temp),"^;Imag;");
45          else
46              error("wrong second argument");
47          endif
48      endif
49
50  else
51      error("PlotAmps takes one or two arguments");
52  endif
53
54  endfunction
