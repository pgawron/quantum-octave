 2  ## @deftypefn {Function file} {} ReverseVec(@var{vec})
 3  ## Function @code{ReverseVec} reverses indexes in vector @var{vec}.
 4  ## 
 5  ## @example 
 6  ## @group
 7  ## ReverseVec([1,2,3,5])
 8  ##  @result{} 
 9  ##  [5,3,2,1]
10  ## @end group
11  ## @end example
12  ##
13  ## @end deftypefn
14  ##
15  ## @seealso {Dec2BinVec, BinVec2Dec, Ket, BuildBinaryVector}
16  ##

18  ## Created: 25 November 2003
19
20  function ret = ReverseVec(vec)
21  if (nargin =! 1 )
22      usage ("ReverseVec (vector)");
23  endif
24  if ( !isvector(vec) )
25      error ("Single vector expected as argument!");
26  endif
27  vs = size(vec)(2);
28  for i = 1:vs
29      ret(i) = vec(vs+1 -i);
30  endfor
31  endfunction
