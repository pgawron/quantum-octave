29  function ret = Ket (binvec)
30  if ( size(binvec)(1) != 1 )
31      error("Error in input vector!");
32  endif
33  num = 0;
34  for i = 1: size(binvec)(2)
35      temp = binvec(i);
36      if (temp == 0 || temp ==1)
37          num += (2^(i-1))*binvec(i);
38      else
39          error("Number %d found in input vector!",temp);
40      endif
41  endfor
42  ret = KetN(num,size(binvec)(2));
43  endfunction
