15  function ret = Observable(obs, str)
16  # TODO check if obs is hermietian and of size 2x2
17  flag = -1;# 0 - vec, 1 - proj
18  if(nargin==1)
19          flag = 1;
20  elseif (nargin == 2)
21          if(strcmp(str,"vec"))
22                  flag=0;
23          elseif(strcmp(str,"proj"))
24                  flag=1;
25          else
26                  usage("Observable(obs[, {\"proj\" | \"vec\"}])");
27          endif
28  endif
29
30  [V,l]=eig(obs);
31
32
33  ret(1).l = l(1,1);
34  ret(2).l = l(2,2);
35  if(flag==1)
36          ret(1).proj = V(:,1)*V(:,1)';
37          ret(2).proj = V(:,2)*V(:,2)';
38  elseif(flag==0)
39          ret(1).vec = V(:,1);
40          ret(2).vec = V(:,2);
41  else
42          error("internal error :-P")
43  endif
44  endfunction
