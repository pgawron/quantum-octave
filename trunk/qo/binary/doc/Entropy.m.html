1  ## -*- texinfo -*-
 2  ## @deftypefn {Function file} {} Entropy(s)
 3  ## Function @code{H} calculates entropy of mixed (or pure) state
 4  ## @var{s} acording to the rule 
 5  ## @code{Entropy(s)=-sum(eig(s)log2(eig(2))}
 6  ## 
 7  ## @example 
 8  ## @group
 9  ## Entropy(MixState(State(Ket([0,0,1]),State(Ket([1,0,0]))))
10  ##  @result{} 
11  ##  1
12  ## @end group
13  ## @group
14  ## Entropy(MaxMix(2))
15  ##  @result{}
16  ##  2
17  ## @end example
18  ##
19  ## @end deftypefn
20  ##
21  ## @seealso {PTrace, PTranspose, Evolve, Negativity}
22  ##
23  ## Author: Piotr Gawron, Jaroslaw Miszczak
24  ## Created: 22 March 2004
25
26  function ret = Entropy(state)
27  ret = 0;
28  tmp = 0;
29  if ( nargin != 1 )
30      usage ("Entropy (state)");
31  else
32      ev = eig(state); # eigenvalues of density matrix
33      nb = size(ev)(1); # number of eigenvalues
34      for  i = 1:nb
35          if ( ev(i) == 0 )
36              ret -= 0;
37          else
38              ret -= ev(i)*log2(ev(i));
39          endif
40      endfor
41  endif
42  endfunction
